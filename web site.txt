<script>
    // --- BURAYA Dƒ∞KKAT: YENƒ∞ API KEY'ƒ∞Nƒ∞ A≈ûAƒûIYA YAPI≈ûTIR ---
    const API_KEY = "BURAYA_YAPƒ∞STƒ∞R_KANALA_DA_ABONE_OLMAYI_UNUTMA"; 
    // √ñrnek: const API_KEY = "gsk_12345...";

    // G√úNCELLENMƒ∞≈û MODEL ƒ∞SMƒ∞ (En yenisi)
    const MODEL_NAME = "llama-3.3-70b-versatile"; 
    
    const API_URL = "https://api.groq.com/openai/v1/chat/completions";

    let conversationHistory = [
        {
            role: "system",
            content: "Sen 'Pocket Dietitian' adƒ±nda, uzman bir diyetisyensin. Sadece beslenme, kalori ve yemek hakkƒ±nda konu≈ü. Kƒ±sa, net ve emojili cevaplar ver. T√ºrk√ße konu≈ü."
        }
    ];

    const chatContainer = document.getElementById('chatContainer');
    const userInput = document.getElementById('userInput');
    const typingIndicator = document.getElementById('typingIndicator');

    // Enter tu≈üuna basƒ±nca √ßalƒ±≈üacak fonksiyon
    function handleEnter(e) {
        if (e.key === 'Enter') {
            sendMessage();
        }
    }

    async function sendMessage() {
        const text = userInput.value.trim();
        if (!text) return;

        // Kullanƒ±cƒ± mesajƒ±nƒ± ekrana bas
        addMessage(text, 'user');
        userInput.value = '';
        
        conversationHistory.push({ role: "user", content: text });

        // Y√ºkleniyor animasyonunu a√ß
        showTyping(true);
        scrollToBottom();

        try {
            const response = await fetch(API_URL, {
                method: "POST",
                headers: {
                    "Authorization": `Bearer ${API_KEY.trim()}`,
                    "Content-Type": "application/json"
                },
                body: JSON.stringify({
                    messages: conversationHistory,
                    model: MODEL_NAME,
                    temperature: 0.5,
                    max_tokens: 256
                })
            });

            const data = await response.json();

            // API Hatasƒ± varsa yakala
            if (!response.ok) {
                showTyping(false);
                const errorMsg = data.error ? data.error.message : "Bilinmeyen Hata";
                alert("API HATASI: " + errorMsg);
                conversationHistory.pop(); // Hatalƒ± mesajƒ± ge√ßmi≈üten sil
                return;
            }

            // Ba≈üarƒ±lƒ± cevap
            if (data.choices && data.choices.length > 0) {
                const aiReply = data.choices[0].message.content;
                showTyping(false);
                addMessage(aiReply, 'ai');
                conversationHistory.push({ role: "assistant", content: aiReply });
            }

        } catch (error) {
            console.error("Baƒülantƒ± Hatasƒ±:", error);
            showTyping(false);
            alert("Bir hata olu≈ütu: " + error);
        }

        scrollToBottom();
    }

    function addMessage(text, sender) {
        const div = document.createElement('div');
        div.className = `message ${sender}`;
        let formattedText = text.replace(/\*\*(.*?)\*\*/g, '<b>$1</b>');
        div.innerHTML = formattedText;
        chatContainer.appendChild(div);
    }

    function showTyping(show) {
        if (show) {
            chatContainer.appendChild(typingIndicator);
            typingIndicator.classList.add('active');
        } else {
            typingIndicator.classList.remove('active');
        }
    }

    function scrollToBottom() {
        chatContainer.scrollTop = chatContainer.scrollHeight;
    }
</script>
